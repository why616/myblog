class Cat{
        constructor(name,age){
            this.name = name;
            this.age = age;
        }//别加,
        //别加function
        Say(){
            return '我的名字是' + this.name;
        }
    }
    var cat1 = new Cat('有鱼',2);
    console.log(cat1.Say());//我的名字是有鱼
    //类本质上是函数，是构造函数的另一种写法
    console.log(typeof Cat);// function
    console.log(Cat.prototype.constructor ===Cat);//true
    //也可以通过原型修改
